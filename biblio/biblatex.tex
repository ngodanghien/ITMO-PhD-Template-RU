%%% Реализация библиографии пакетами biblatex и biblatex-gost с использованием движка biber %%%

\usepackage{csquotes} % biblatex рекомендует его подключать

\usepackage[%
    backend=biber,
    bibencoding=utf8,
    sorting=none,
    style=gost-numeric,
    language=autobib,
    autolang=other,
    clearlang=true,
    defernumbers=true,
    sortcites=true,
    doi=false,
    isbn=false,
]{biblatex}

\addbibresource{biblio/references.bib}
